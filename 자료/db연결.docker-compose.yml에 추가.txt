import pandas as pd
from sqlalchemy import create_engine

# DB 연결
engine = create_engine("mysql+pymysql://root:15932!miniprojectdb@192.168.0.41:30000/miniproject")

# CSV 읽기
df = pd.read_csv(r"D:\github\miniproject\문화체육관광부_전국 골프장 현황_20221231.csv")

# DB에 저장 (id는 자동 생성되게, DataFrame엔 없음)
df.to_sql("glof", con=engine, if_exists="replace", index=False)

# id 컬럼 자동 증가 추가 (필요시 ALTER)
with engine.connect() as conn:
    conn.execute("""
        ALTER TABLE glof
        ADD COLUMN id INT PRIMARY KEY AUTO_INCREMENT FIRST;
    """)
